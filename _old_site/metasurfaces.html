Flat Optics Revolution

Complete Wavefront Control at Subwavelength Scale

Metasurfaces consist of carefully designed arrays of subwavelength nanostructures that locally control the amplitude, phase, and polarization of light. By engineering the geometry and arrangement of these meta-atoms, we can realize any optical function in an ultrathin form factor.

metalens_design.py 
import metadesign as md
from metadesign.materials import TiO2, SiO2
from metadesign.objectives import FocalLength, Efficiency, Aberration
from metadesign.optimization import GradientBased, Evolutionary

# Define metalens specifications
metalens = md.Metalens(
    diameter=10e-3,          # 10 mm diameter
    focal_length=50e-3,      # 50 mm focal length
    wavelength=550e-9,       # 550 nm design wavelength
    numerical_aperture=0.8   # High NA
)

# Create metasurface with TiO2 nanopillars on glass
metasurface = md.Metasurface(
    substrate=SiO2(),
    meta_atom=md.Nanopillar(
        material=TiO2(),
        height=600e-9,       # 600 nm height
        radius_range=(50e-9, 200e-9),  # 50-200 nm radius
        lattice_constant=300e-9         # 300 nm pitch
    )
)

# Define optimization objectives
objectives = [
    FocalLength(target=50e-3, weight=1.0),
    Efficiency(target=0.95, weight=0.8),
    Aberration(max_rms=0.1, weight=0.5)
]

# Run inverse design optimization
optimizer = GradientBased(
    method='adjoint',
    learning_rate=0.01,
    iterations=1000
)

# Optimize metasurface design
optimized_design = optimizer.optimize(
    metasurface=metasurface,
    objectives=objectives,
    constraints=[
        md.Constraint('min_feature_size', 50e-9),
        md.Constraint('aspect_ratio', max_value=10)
    ]
)

# Simulate performance
simulator = md.RCWA(
    harmonics=21,
    polarization='circular',
    gpu_acceleration=True
)

results = simulator.analyze(optimized_design, wavelengths=np.linspace(400e-9, 700e-9, 100))

# Extract key metrics
print(f"Focusing efficiency: {results.efficiency:.1%}")
print(f"Strehl ratio: {results.strehl_ratio:.3f}")
print(f"Chromatic focal shift: {results.chromatic_shift*1e3:.1f} mm")

# Generate hologram for arbitrary beam shaping
hologram = md.Hologram(
    target_image='ABP_logo.png',
    reconstruction_distance=1.0,  # 1 meter
    encoding='amplitude_phase'
)

# Design holographic metasurface
holo_meta = hologram.generate_metasurface(
    metasurface_template=metasurface,
    optimization_method='gerchberg_saxton',
    iterations=100
)

# Export for fabrication
holo_meta.export_gds('holographic_metasurface.gds', layer=1)
holo_meta.export_dose_map('ebeam_doses.txt', compensation=True)
