import torch
import torch.nn as nn
from transformers import AutoModel

class MYNXPredictor(nn.Module):
    def __init__(self):
        super().__init__()
        self.clinical_encoder = ClinicalDataEncoder()
        self.imaging_encoder = ImagingCNN3D()
        self.genomic_encoder = GenomicTransformer()
        self.fusion_network = MultiModalFusion()
        
    def forward(self, clinical, imaging, genomic):
        clinical_features = self.clinical_encoder(clinical)
        imaging_features = self.imaging_encoder(imaging)
        genomic_features = self.genomic_encoder(genomic)
        
        fused = self.fusion_network(
            clinical_features,
            imaging_features,
            genomic_features
        )
        
        return self.predict_risks(fused)
