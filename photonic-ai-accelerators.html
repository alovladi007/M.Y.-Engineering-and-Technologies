Quick Start Code

# Install PhotonFlow runtime
pip install photonflow-runtime

# Initialize device
import photonflow as pf
device = pf.Device('ONN-Cloud-1000')

# Load pre-trained model
model = pf.load_model('bert-large-optical')
model.to(device)

# Run inference
output = model(input_data)
print(f"Latency: {model.last_latency_ms:.3f} ms")
