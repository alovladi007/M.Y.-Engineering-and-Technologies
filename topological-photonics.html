topological_waveguide.py 
import topoflow as tf
from topoflow.crystals import HoneycombLattice
from topoflow.materials import MagnetoOptic
from topoflow.solvers import TopologicalFDTD

# Create honeycomb photonic crystal with broken time-reversal symmetry
lattice = HoneycombLattice(
    lattice_constant=500e-9,  # 500 nm
    radius=0.3,              # r/a ratio
    material=MagnetoOptic(
        epsilon=12.0,
        gyration=0.1,        # Magneto-optic coefficient
        loss=0.001
    )
)

# Design topological interface
interface = tf.TopologicalInterface(
    crystal_A=lattice.with_gyration(+0.1),  # Positive gyration
    crystal_B=lattice.with_gyration(-0.1),  # Negative gyration
    interface_type='zigzag'
)

# Calculate band structure and topological invariants
bands = interface.calculate_bands(
    k_path=['Γ', 'K', 'M', 'Γ'],
    num_bands=20
)

# Extract Chern numbers
chern_numbers = bands.calculate_chern_numbers()
print(f"Band gap Chern number: {chern_numbers[8] - chern_numbers[7]}")

# Simulate edge state propagation
solver = TopologicalFDTD(
    structure=interface,
    resolution=20,  # Points per lattice constant
    boundary_conditions='PML'
)

# Launch edge state
source = tf.EdgeStateSource(
    wavelength=1550e-9,
    position='interface',
    direction='+x'
)

# Run simulation with disorder
solver.add_disorder(
    type='positional',
    strength=0.1  # 10% random displacement
)

fields = solver.run(
    source=source,
    time_steps=5000,
    monitors=[
        tf.PowerMonitor('transmission'),
        tf.FieldMonitor('edge_propagation')
    ]
)

# Analyze robustness
transmission = fields.monitors['transmission'].get_transmission()
print(f"Transmission through disordered region: {transmission:.2%}")

# Visualize edge state confinement
edge_field = fields.monitors['edge_propagation'].get_intensity()
tf.visualize.plot_edge_state(edge_field, interface)
